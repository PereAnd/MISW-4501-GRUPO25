    --policy-name AWSLoadBalancerControllerIAMPolicy7 \

  --cluster=backend-proyecto6 \
  --namespace=kube-system \
  --name=aws-load-balancer-controller \
  --role-name AmazonEKSLoadBalancerControllerRole7 \
  --attach-policy-arn=arn:aws:iam::798178280816:policy/AWSLoadBalancerControllerIAMPolicy7 \
  --approve --override-existing-serviceaccounts

  -n kube-system \
  --set clusterName=backend-proyecto6 \
  --set serviceAccount.create=false \
